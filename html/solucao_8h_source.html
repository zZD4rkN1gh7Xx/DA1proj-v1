106bool without_bfs(WMSGraph& residual, Vertex<Agua> remove, Vertex<Agua>& source, Vertex<Agua>& sink, std::unordered_map<std::string, Vertex<Agua>*>& parent) {
112 for (auto& v : residual.getVertexSet()) {
113 v->setVisited(false);
114 }
115
116 std::queue<Vertex<Agua>*> q;
117 q.push(&source);
118 source.setVisited(true);
119
120 while (!q.empty()) {
121 Vertex<Agua>* u = q.front();
122 q.pop();
123
124 for (auto& edge : u->getAdj()) {
125 if (!edge.getDest()->isVisited() && residual.get_pipe_id(residual.findEdge(u->getInfo(), edge.getDest()->getInfo())->getWeight().get_id()).get_capacity() > 0 && edge.getDest()->getInfo().get_code() != remove.getInfo().get_code()) {
126 q.push(edge.getDest());
127 parent[edge.getDest()->getInfo().get_code()] = u;
128 edge.getDest()->setVisited(true);
